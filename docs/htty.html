551    """
552    Run a command using the 'ht' tool and return a HtWrapper object.
553
554    This version uses --wait-for-output by default to avoid race conditions.
555    """
556    # Use provided logger or fall back to default
557    process_logger = logger or default_logger
558
559    # Create a queue for events
560    event_queue: queue.Queue[dict[str, Any]] = queue.Queue()
561
562    # Build the ht subscription list
563    base_subscribes = [
564        HtEvent.INIT,
565        HtEvent.SNAPSHOT,
566        HtEvent.OUTPUT,
567        HtEvent.RESIZE,
568        HtEvent.PID,
569        HtEvent.EXIT_CODE,
570        HtEvent.COMMAND_COMPLETED,
571    ]
572    if extra_subscribes:
573        # Convert string subscribes to HtEvent enum values
574        for sub in extra_subscribes:
575            try:
576                base_subscribes.append(HtEvent(sub))
577            except ValueError:
578                process_logger.warning(f"Unknown subscription event: {sub}")
579
580    # Convert command to string if it's a list
581    command_str = command if isinstance(command, str) else " ".join(command)
582
583    # Create HtArgs and use htty_core.run()
584    ht_args = HtArgs(
585        command=command_str,  # Use the already-formatted command string
586        subscribes=base_subscribes,
587        rows=rows,
588        cols=cols,
589    )
590
591    # Log the exact command that would be run
592    cmd_args = ht_args.get_command()
593    process_logger.debug(f"Launching command: {' '.join(cmd_args)}")
594
595    ht_proc = htty_core_run(ht_args)
596
597    process_logger.debug(f"ht started: PID {ht_proc.pid}")

599    # Create a reader thread to capture ht output
600    def reader_thread(
601        ht_proc: subprocess.Popen[str],
602        queue_obj: queue.Queue[dict[str, Any]],
603        ht_process: HtWrapper,
604        thread_logger: logging.Logger,
605    ) -> None:
606        thread_logger.debug(f"Reader thread started for ht process {ht_proc.pid}")
607
608        while True:
609            if ht_proc.stdout is None:
610                thread_logger.warning(f"ht process {ht_proc.pid} stdout is None, exiting reader thread")
611                break
612
613            line = ht_proc.stdout.readline()
614            if not line:
615                thread_logger.debug(f"ht process {ht_proc.pid} stdout closed, exiting reader thread")
616                break
617
618            line = line.strip()
619            if not line:
620                continue
621
622            try:
623                event = json.loads(line)
624                thread_logger.debug(f"ht event: {event}")
625                queue_obj.put(event)
626
627                if event["type"] == "output":
628                    ht_process.add_output_event(event)
629                elif event["type"] == "exitCode":
630                    thread_logger.debug(
631                        f"ht process {ht_proc.pid} subprocess exited with code: {event.get('data', {}).get('exitCode')}"
632                    )
633                    ht_process.set_subprocess_exited(True)
634                    exit_code = event.get("data", {}).get("exitCode")
635                    if exit_code is not None:
636                        ht_process.cmd.exit_code = exit_code
637                elif event["type"] == "pid":
638                    thread_logger.debug(f"ht process {ht_proc.pid} subprocess PID: {event.get('data', {}).get('pid')}")
639                    pid = event.get("data", {}).get("pid")
640                    if pid is not None:
641                        ht_process.cmd.pid = pid
642                elif event["type"] == "commandCompleted":
643                    # Command has completed - this is the reliable signal that subprocess finished
644                    ht_process.set_subprocess_completed(True)
645                elif event["type"] == "debug":
646                    thread_logger.debug(f"ht process {ht_proc.pid} debug: {event.get('data', {})}")
647                    # Note: We no longer rely on debug events for subprocess_completed
648                    # The commandCompleted event (above) is the reliable source
649            except json.JSONDecodeError as e:
650                # Only log raw stdout when we can't parse it as JSON - this indicates an unexpected message
651                thread_logger.warning(f"ht process {ht_proc.pid} non-JSON stdout: {line} (error: {e})")
652                pass
653
654        thread_logger.debug(f"Reader thread exiting for ht process {ht_proc.pid}")
655
656    # Create an HtWrapper instance
657    process = HtWrapper(
658        ht_proc,
659        event_queue,
660        command=command_str,
661        rows=rows,
662        cols=cols,
663        no_exit=no_exit,
664        logger=process_logger,
665    )
666
667    # Start the reader thread for stdout
668    stdout_thread = threading.Thread(
669        target=reader_thread,
670        args=(ht_proc, event_queue, process, process_logger),
671        daemon=True,
672    )
673    stdout_thread.start()
674
675    # Start a stderr reader thread
676    def stderr_reader_thread(ht_proc: subprocess.Popen[str], thread_logger: logging.Logger) -> None:
677        thread_logger.debug(f"Stderr reader thread started for ht process {ht_proc.pid}")
678
679        while True:
680            if ht_proc.stderr is None:
681                thread_logger.warning(f"ht process {ht_proc.pid} stderr is None, exiting stderr reader thread")
682                break
683
684            line = ht_proc.stderr.readline()
685            if not line:
686                thread_logger.debug(f"ht process {ht_proc.pid} stderr closed, exiting stderr reader thread")
687                break
688
689            line = line.strip()
690            if line:
691                thread_logger.debug(f"ht stderr: {line}")
692
693        thread_logger.debug(f"Stderr reader thread exiting for ht process {ht_proc.pid}")
694
695    stderr_thread = threading.Thread(target=stderr_reader_thread, args=(ht_proc, process_logger), daemon=True)
696    stderr_thread.start()
697
698    # Wait briefly for the process to initialize and get PID
699    start_time = time.time()
700    while time.time() - start_time < 2:
701        try:
702            event = event_queue.get(block=True, timeout=0.5)
703            if event["type"] == "pid":
704                pid = event["data"]["pid"]
705                process.cmd.pid = pid
706                break
707        except queue.Empty:
708            continue
709
710    time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
711    return process

216        """
217        if self._ht_proc.poll() is not None:
218            raise RuntimeError(f"ht process has exited with code {self._ht_proc.returncode}")
219
220        message = json.dumps({"type": "takeSnapshot"})
221        self._logger.debug(f"Taking snapshot: {message}")
222
223        try:
224            if self._ht_proc.stdin is not None:
225                self._ht_proc.stdin.write(message + "\n")
226                self._ht_proc.stdin.flush()

230        except BrokenPipeError as e:

232            self._logger.error(f"ht process poll result: {self._ht_proc.poll()}")

236            ) from e

239
240        # Process events until we find the snapshot
241        retry_count = 0
242        while retry_count < MAX_SNAPSHOT_RETRIES:

244                event = self._event_queue.get(block=True, timeout=SNAPSHOT_RETRY_TIMEOUT)
245            except queue.Empty:
246                retry_count += 1
247                continue
248
249            if event["type"] == "snapshot":
250                data = event["data"]
251                snapshot_text = data["text"]

266                if "rows" in data:

269                    self._cols = data["cols"]
270            elif event["type"] == "init":
271                pass

273                # Put non-snapshot events back in queue for reader thread to handle
274                self._event_queue.put(event)

281    def exit(self, timeout: float = DEFAULT_EXIT_TIMEOUT) -> int:
282        """
283        Exit the ht process, ensuring clean shutdown.
284
285        Uses different strategies based on subprocess state:
286        - If subprocess already exited (exitCode event received): graceful shutdown via exit command
287        - If subprocess still running: forced termination with SIGTERM then SIGKILL
288        """
289        self._logger.debug(f"Exiting HTProcess: ht_proc.pid={self._ht_proc.pid}")
290
291        # Check if we've already received the exitCode event
292        if self._subprocess_exited:
293            self._logger.debug("Subprocess already exited (exitCode event received), attempting graceful shutdown")
294            return self._graceful_exit(timeout)
295        else:
296            self._logger.debug("Subprocess has not exited yet, checking current state")
297
298            # Give a brief moment for any pending exitCode event to arrive
299            brief_wait_start = time.time()
300            while time.time() - brief_wait_start < 0.5:  # Wait up to 500ms
301                if self._subprocess_exited:
302                    self._logger.debug("Subprocess exited during brief wait, attempting graceful shutdown")
303                    return self._graceful_exit(timeout)
304                time.sleep(0.01)
305
306            self._logger.debug("Subprocess still running after brief wait, using forced termination")
307            return self._forced_exit(timeout)

309    def _graceful_exit(self, timeout: float) -> int:
310        """
311        Graceful exit: subprocess has completed, so we can send exit command to ht process.
312        """
313        # Send exit command to ht process
314        message = json.dumps({"type": "exit"})
315        self._logger.debug(f"Sending exit command to ht process {self._ht_proc.pid}: {message}")
316
317        try:
318            if self._ht_proc.stdin is not None:
319                self._ht_proc.stdin.write(message + "\n")
320                self._ht_proc.stdin.flush()
321                self._logger.debug(f"Exit command sent successfully to ht process {self._ht_proc.pid}")
322                self._ht_proc.stdin.close()  # Close stdin after sending exit command
323                self._logger.debug(f"Closed stdin for ht process {self._ht_proc.pid}")
324            else:
325                self._logger.debug(f"ht process {self._ht_proc.pid} stdin is None, cannot send exit command")
326        except (BrokenPipeError, OSError) as e:
327            self._logger.debug(
328                f"Failed to send exit command to ht process {self._ht_proc.pid}: {e} (process may have already exited)"
329            )
330            pass
331
332        # Wait for the ht process to finish gracefully
333        start_time = time.time()
334        while self._ht_proc.poll() is None:
335            if time.time() - start_time > timeout:
336                # Graceful exit timed out, fall back to forced termination
337                self._logger.warning(
338                    f"ht process {self._ht_proc.pid} did not exit gracefully within timeout, "
339                    f"falling back to forced termination"
340                )
341                return self._forced_exit(timeout)
342            time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
343
344        self._exit_code = self._ht_proc.returncode
345        if self._exit_code is None:
346            raise RuntimeError("Failed to determine ht process exit code")
347
348        self._logger.debug(f"HTProcess exited gracefully: exit_code={self._exit_code}")
349        return self._exit_code

351    def _forced_exit(self, timeout: float) -> int:
352        """
353        Forced exit: subprocess may still be running, so we need to terminate everything forcefully.
354        """
355        # Step 1: Ensure subprocess is terminated first if needed
356        if self._cmd_process.pid and not self._subprocess_exited:
357            self._logger.debug(f"Terminating subprocess: pid={self._cmd_process.pid}")
358            try:
359                os.kill(self._cmd_process.pid, 0)
360                self._cmd_process.terminate()
361                try:
362                    self._cmd_process.wait(timeout=DEFAULT_SUBPROCESS_WAIT_TIMEOUT)
363                    self._logger.debug(f"Subprocess {self._cmd_process.pid} terminated successfully")
364                except Exception:
365                    self._logger.warning(f"Subprocess {self._cmd_process.pid} did not terminate gracefully, killing")
366                    with suppress(Exception):
367                        self._cmd_process.kill()
368            except OSError:
369                self._logger.debug(f"Subprocess {self._cmd_process.pid} already exited")
370                pass  # Process already exited
371
372        # Step 2: Force terminate the ht process with SIGTERM, then SIGKILL if needed
373        self._logger.debug(f"Force terminating ht process {self._ht_proc.pid}")
374
375        # Try SIGTERM first
376        try:
377            self._ht_proc.terminate()
378            self._logger.debug(f"Sent SIGTERM to ht process {self._ht_proc.pid}")
379        except Exception as e:
380            self._logger.debug(f"Failed to send SIGTERM to ht process {self._ht_proc.pid}: {e}")
381
382        # Wait for termination
383        start_time = time.time()
384        while self._ht_proc.poll() is None:
385            if time.time() - start_time > timeout:
386                # SIGTERM timeout, try SIGKILL
387                self._logger.warning(
388                    f"ht process {self._ht_proc.pid} did not terminate with SIGTERM within timeout, sending SIGKILL"
389                )
390                try:
391                    self._ht_proc.kill()
392                    self._logger.debug(f"Sent SIGKILL to ht process {self._ht_proc.pid}")
393                except Exception as e:
394                    self._logger.debug(f"Failed to send SIGKILL to ht process {self._ht_proc.pid}: {e}")
395
396                # Wait for SIGKILL to take effect
397                kill_start_time = time.time()
398                while self._ht_proc.poll() is None:
399                    if time.time() - kill_start_time > timeout:
400                        self._logger.error(f"ht process {self._ht_proc.pid} did not respond to SIGKILL within timeout")
401                        break
402                    time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
403                break
404            time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
405
406        self._exit_code = self._ht_proc.returncode
407        if self._exit_code is None:
408            raise RuntimeError("Failed to determine ht process exit code")
409
410        self._logger.debug(f"HTProcess exited via forced termination: exit_code={self._exit_code}")
411        return self._exit_code

413    def expect(self, pattern: str, timeout: float = DEFAULT_EXPECT_TIMEOUT) -> None:
414        """
415        Wait for a regex pattern to appear in the terminal output.
416
417        This method efficiently waits for output by monitoring the output events from
418        the ht process rather than polling with snapshots. It checks both the current
419        terminal state (via snapshot) and any new output that arrives.
420
421        Args:
422            pattern: The regex pattern to look for in the terminal output
423            timeout: Maximum time to wait in seconds (default: 5.0)
424
425        Raises:
426            TimeoutError: If the pattern doesn't appear within the timeout period
427            RuntimeError: If the ht process has exited
428        """
429        if self._ht_proc.poll() is not None:
430            raise RuntimeError(f"ht process has exited with code {self._ht_proc.returncode}")
431
432        self._logger.debug(f"Expecting regex pattern: '{pattern}'")
433
434        # Compile the regex pattern
435        try:
436            regex = re.compile(pattern)
437        except re.error as e:
438            raise ValueError(f"Invalid regex pattern '{pattern}': {e}") from e
439
440        # First check current terminal state
441        snapshot = self.snapshot()
442        if regex.search(snapshot.text):
443            self._logger.debug(f"Pattern '{pattern}' found immediately in current terminal state")
444            return
445
446        # Start time for timeout tracking
447        start_time = time.time()
448
449        # Process events until we find the pattern or timeout
450        while True:
451            # Check timeout
452            if time.time() - start_time > timeout:
453                self._logger.debug(f"Pattern '{pattern}' not found in terminal output after {timeout} seconds")
454                raise TimeoutError(f"Pattern '{pattern}' not found within {timeout} seconds")
455
456            try:
457                # Wait for next event with a short timeout to allow checking the overall timeout
458                event = self._event_queue.get(block=True, timeout=0.1)
459            except queue.Empty:
460                continue
461
462            # Process the event
463            if event["type"] == "output":
464                self._output_events.append(event)
465                # Check if pattern appears in this output
466                if "data" in event and "seq" in event["data"] and regex.search(event["data"]["seq"]):
467                    self._logger.debug(f"Pattern '{pattern}' found in output event")
468                    return
469            elif event["type"] == "exitCode":
470                # Put back in queue for reader thread to handle
471                self._event_queue.put(event)
472                # Don't raise here - the process might have exited after outputting what we want
473            elif event["type"] == "snapshot":
474                # If we get a snapshot event, check its content
475                if "data" in event and "text" in event["data"] and regex.search(event["data"]["text"]):
476                    self._logger.debug(f"Pattern '{pattern}' found in snapshot event")
477                    return
478
479            # Take a new snapshot periodically to catch any missed output
480            if len(self._output_events) % 10 == 0:  # Every 10 events
481                snapshot = self.snapshot()
482                if regex.search(snapshot.text):
483                    self._logger.debug(f"Pattern '{pattern}' found in periodic snapshot")
484                    return

416
417        This method efficiently waits for output by monitoring the output events from
418        the ht process rather than polling with snapshots. It checks both the current

422            pattern: The regex pattern to look for in the terminal output
423            timeout: Maximum time to wait in seconds (default: 5.0)
424
425        Raises:
426            TimeoutError: If the pattern doesn't appear within the timeout period
427            RuntimeError: If the ht process has exited
428        """
429        if self._ht_proc.poll() is not None:
430            raise RuntimeError(f"ht process has exited with code {self._ht_proc.returncode}")
431
432        self._logger.debug(f"Expecting regex pattern: '{pattern}'")
433
434        # Compile the regex pattern
435        try:
436            regex = re.compile(pattern)
437        except re.error as e:
438            raise ValueError(f"Invalid regex pattern '{pattern}': {e}") from e
439
440        # First check current terminal state
441        snapshot = self.snapshot()
442        if regex.search(snapshot.text):
443            self._logger.debug(f"Pattern '{pattern}' found immediately in current terminal state")
444            return
445
446        # Start time for timeout tracking
447        start_time = time.time()
448
449        # Process events until we find the pattern or timeout
450        while True:
451            # Check timeout
452            if time.time() - start_time > timeout:
453                self._logger.debug(f"Pattern '{pattern}' not found in terminal output after {timeout} seconds")
454                raise TimeoutError(f"Pattern '{pattern}' not found within {timeout} seconds")
455
456            try:
457                # Wait for next event with a short timeout to allow checking the overall timeout
458                event = self._event_queue.get(block=True, timeout=0.1)
459            except queue.Empty:
460                continue
461
462            # Process the event
463            if event["type"] == "output":
464                self._output_events.append(event)
465                # Check if pattern appears in this output

468                    return

471                self._event_queue.put(event)
472                # Don't raise here - the process might have exited after outputting what we want
473            elif event["type"] == "snapshot":
474                # If we get a snapshot event, check its content

477                    return

482                if regex.search(snapshot.text):
483                    self._logger.debug(f"Pattern '{pattern}' found in periodic snapshot")
484                    return
