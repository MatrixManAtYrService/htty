 15def get_ht_binary():
 16    """
 17    Get the path to the ht binary.
 18
 19    Order of precedence:
 20    1. HTUTIL_HT_BIN environment variable (if set and valid) - user override
 21    2. Bundled ht binary via importlib.resources (production/wheel)
 22    3. System 'ht' command from PATH (development fallback)
 23
 24    Raises:
 25        RuntimeError: If ht binary cannot be found anywhere
 26    """
 27    import os
 28    import shutil
 29    from pathlib import Path
 30    import logging
 31
 32    logger = logging.getLogger(__name__)
 33
 34    # 1. Check for user-specified ht binary override
 35    user_ht = os.environ.get("HTUTIL_HT_BIN")
 36    if user_ht and user_ht.strip():  # Ignore empty values
 37        user_ht_path = Path(user_ht)
 38        if user_ht_path.is_file() and os.access(str(user_ht_path), os.X_OK):
 39            logger.info("Using user-specified ht binary from HTUTIL_HT_BIN")
 40            return str(user_ht_path)
 41        else:
 42            raise RuntimeError(
 43                f"HTUTIL_HT_BIN='{user_ht}' is not a valid executable file. "
 44                f"Please check that the path exists and is executable."
 45            )
 46
 47    # 2. Check for bundled ht binary using importlib.resources (production)
 48    try:
 49        from importlib import resources as impresources
 50        from . import _bundled
 51        import tempfile
 52        import atexit
 53
 54        bundled_files = impresources.files(_bundled)
 55        ht_resource = bundled_files / "ht"
 56
 57        if ht_resource.is_file():
 58            logger.info("Using bundled ht binary")
 59            # Read the binary data
 60            ht_data = ht_resource.read_bytes()
 61
 62            # Create a temporary executable file
 63            with tempfile.NamedTemporaryFile(
 64                mode="wb", prefix="ht_", delete=False
 65            ) as tmp_file:
 66                tmp_file.write(ht_data)
 67                tmp_path = tmp_file.name
 68
 69            # Make it executable
 70            os.chmod(tmp_path, 0o755)
 71
 72            # Register cleanup on exit
 73            atexit.register(
 74                lambda: os.unlink(tmp_path) if os.path.exists(tmp_path) else None
 75            )
 76
 77            return tmp_path
 78
 79    except (ImportError, FileNotFoundError, AttributeError):
 80        # No bundled binary available - this is normal in development mode
 81        logger.debug("No bundled ht binary found, falling back to system PATH")
 82
 83    # 3. Fall back to system 'ht' command (development mode)
 84    system_ht = shutil.which("ht")
 85    if system_ht:
 86        logger.warning(
 87            f"Using system ht binary from PATH: {system_ht}. "
 88            "Version and compatibility are not guaranteed. "
 89            "Consider using a wheel distribution with bundled ht for production."
 90        )
 91        return system_ht
 92
 93    # 4. If we get here, ht is not available anywhere
 94    raise RuntimeError(
 95        "Could not find ht binary. Please ensure one of the following:\n"
 96        "1. Install htutil from a wheel distribution (includes bundled ht)\n"
 97        "2. Install the 'ht' tool separately and ensure it's in PATH\n"
 98        "3. Set HTUTIL_HT_BIN environment variable to point to ht binary\n"
 99        "   Example: export HTUTIL_HT_BIN=/path/to/ht\n"
100        "Get ht from: https://github.com/andyk/ht"
101    )

104def clean_ansi_for_html(ansi_text: str) -> str:
105    """
106    Clean ANSI sequences to keep only color/style codes that ansi2html can handle.
107    Removes cursor positioning, screen control, and other non-display sequences.
108    """
109    import re
110
111    # First, normalize \x9b sequences to \x1b[ sequences for consistency
112    ansi_text = ansi_text.replace("\x9b", "\x1b[")
113
114    # Remove cursor positioning sequences like \x1b[1;1H, \x1b[2;3H etc.
115    ansi_text = re.sub(r"\x1b\[\d*;\d*H", "", ansi_text)
116
117    # Remove single cursor positioning like \x1b[H
118    ansi_text = re.sub(r"\x1b\[H", "", ansi_text)
119
120    # Remove screen buffer switching \x1b[?1047h, \x1b[?1047l
121    ansi_text = re.sub(r"\x1b\[\?\d+[hl]", "", ansi_text)
122
123    # Remove scroll region setting \x1b[1;4r
124    ansi_text = re.sub(r"\x1b\[\d*;\d*r", "", ansi_text)
125
126    # Remove save/restore cursor sequences \x1b7, \x1b8
127    ansi_text = re.sub(r"\x1b[78]", "", ansi_text)
128
129    # Remove other terminal control sequences but keep color codes
130    # This removes sequences that don't end with 'm' (which are color codes)
131    ansi_text = re.sub(r"\x1b\[(?![0-9;]*m)[^m]*[a-zA-Z]", "", ansi_text)
132
133    # Remove control characters but preserve \x1b which is needed for ANSI codes
134    # and preserve \r\n for line breaks
135    ansi_text = re.sub(r"[\x00-\x08\x0B-\x1A\x1C-\x1F\x7F-\x9F]", "", ansi_text)
136
137    return ansi_text

148class SubprocessController:
149    """Controller for the subprocess being monitored by ht."""
150
151    def __init__(self, pid: Optional[int] = None):
152        self.pid = pid
153        self.exit_code: Optional[int] = None
154
155    def terminate(self) -> None:
156        """Terminate the subprocess."""
157        if self.pid is None:
158            raise RuntimeError("No subprocess PID available")
159        try:
160            os.kill(self.pid, signal.SIGTERM)
161        except OSError:
162            # Process may have already exited
163            pass
164
165    def kill(self) -> None:
166        """Force kill the subprocess."""
167        if self.pid is None:
168            raise RuntimeError("No subprocess PID available")
169        try:
170            os.kill(self.pid, signal.SIGKILL)
171        except OSError:
172            # Process may have already exited
173            pass
174
175    def wait(self, timeout: Optional[float] = None) -> Optional[int]:
176        """
177        Wait for the subprocess to finish.
178
179        Args:
180            timeout: Maximum time to wait (in seconds). If None, waits indefinitely.
181
182        Returns:
183            The exit code of the subprocess, or None if timeout reached
184        """
185        if self.pid is None:
186            raise RuntimeError("No subprocess PID available")
187
188        start_time = time.time()
189        while True:
190            try:
191                # Check if the subprocess is still running
192                os.kill(self.pid, 0)
193                # Process is still running
194
195                # Check timeout
196                if timeout is not None and (time.time() - start_time) > timeout:
197                    return None  # Timeout reached
198
199                time.sleep(0.1)
200
201            except OSError:
202                # Process no longer exists (finished)
203                self.exit_code = 0  # We can't easily get the real exit code
204                return self.exit_code

249        self.cols = cols

251        self.subprocess_exited = False  # Track if subprocess has exited

256        # Debug: let's see what we actually have
257        if not self.output_events:
258            return []
259        return [event for event in self.output_events if event.get("type") == "output"]

261    def send_keys(self, keys: Union[KeyInput, List[KeyInput]]) -> None:
262        """
263        Send keys to the terminal.
264
265        Args:
266            keys: A string, Press enum, or list of keys to send.

280        self.proc.stdin.write(

286    def snapshot(self, timeout: float = 5.0) -> SnapshotResult:
287        """
288        Take a snapshot of the terminal output.
289
290        Returns:
291            SnapshotResult with text (plain), html (styled), and raw_seq (ANSI codes)
292        """
293        # Check if the ht process is still running
294        if self.proc.poll() is not None:
295            raise RuntimeError(
296                f"ht process has exited with code {self.proc.returncode}"
297            )
298
299        try:
300            self.proc.stdin.write(json.dumps({"type": "takeSnapshot"}) + "\n")
301            self.proc.stdin.flush()
302        except BrokenPipeError as e:
303            raise RuntimeError(
304                f"Cannot communicate with ht process (broken pipe). Process may have exited. Poll result: {self.proc.poll()}"
305            ) from e
306
307        sleep(0.1)
308
309        # Process events until we find the snapshot, but don't discard other events
310        while True:
311            event = self.event_queue.get(block=True, timeout=0.5)
312
313            if event["type"] == "snapshot":
314                data = event["data"]
315                snapshot_text = data["text"]
316                raw_seq = data["seq"]
317
318                # Clean the ANSI sequences for HTML conversion
319                cleaned_seq = clean_ansi_for_html(raw_seq)
320
321                # Convert cleaned ANSI sequences to HTML
322                conv = Ansi2HTMLConverter()
323                html = conv.convert(cleaned_seq)
324
325                return SnapshotResult(
326                    text=snapshot_text,
327                    html=html,
328                    raw_seq=raw_seq,  # Keep original raw sequence
329                )
330            elif event["type"] == "output":
331                # Don't lose output events - store them properly
332                self.output_events.append(event)
333            # For other event types (resize, pid, etc.), we could handle them here too
334            # For now, we'll just continue to avoid losing the snapshot event
335
336    def exit(self, timeout: float = 5.0) -> int:
337        """
338        Exit the ht process, forcefully terminating the subprocess if needed.
339
340        This method ensures a reliable exit regardless of subprocess state:
341        - If subprocess is still running, it will be terminated first
342        - Then the ht process will be cleanly shut down
343
344        Args:
345            timeout: Maximum time to wait for the process to exit (default: 5 seconds)
346
347        Returns:
348            The exit code of the ht process (0 for success)
349        """
350
351        # Step 1: Ensure subprocess is terminated first
352        if self.subprocess.pid:
353            try:
354                # Check if subprocess is still running
355                os.kill(self.subprocess.pid, 0)
356                # If we get here, subprocess is still running - terminate it
357                self.subprocess.terminate()
358                try:
359                    self.subprocess.wait(timeout=2.0)
360                except Exception:
361                    # If graceful termination fails, force kill
362                    try:
363                        self.subprocess.kill()
364                    except Exception:
365                        pass
366            except OSError:
367                # Subprocess already dead, that's fine
368                pass
369
370        # Step 2: Handle ht process exit
371        if self.no_exit:
372            # Now that subprocess is dead, ht should be waiting for exit Enter
373            # Give it a moment to detect subprocess exit and show the prompt
374            time.sleep(0.2)
375            self.send_keys(Press.ENTER)
376            time.sleep(0.1)
377
378        # Step 3: Wait for the ht process itself to finish with timeout
379        try:
380            start_time = time.time()
381            while self.proc.poll() is None:
382                if time.time() - start_time > timeout:
383                    # Timeout reached, force terminate
384                    self.proc.terminate()
385                    time.sleep(0.1)
386                    if self.proc.poll() is None:
387                        self.proc.kill()
388                    break
389                time.sleep(0.1)
390
391            self.exit_code = self.proc.returncode
392            # Normalize exit code to 0 for successful termination
393            if self.exit_code is None or self.exit_code == -15:  # SIGTERM
394                self.exit_code = 0
395
396        except Exception:
397            # If anything goes wrong, assume success
398            self.exit_code = 0
399
400        return self.exit_code
401
402    def terminate(self) -> None:
403        """Terminate the ht process itself."""
404        try:
405            self.proc.terminate()
406        except Exception:
407            pass
408
409    def kill(self) -> None:
410        """Force kill the ht process itself."""
411        try:
412            self.proc.kill()
413        except Exception:
414            pass
415
416    def wait(self, timeout: Optional[float] = None) -> Optional[int]:
417        """
418        Wait for the ht process itself to finish.
419
420        Args:
421            timeout: Maximum time to wait (in seconds). If None, waits indefinitely.
422
423        Returns:
424            The exit code of the ht process, or None if timeout reached
425        """
426        try:
427            if timeout is None:
428                self.exit_code = self.proc.wait()
429            else:
430                self.exit_code = self.proc.wait(timeout=timeout)
431            return self.exit_code
432        except subprocess.TimeoutExpired:
433            return None
434        except Exception:
435            return None

438def run(
439    command: str,
440    rows: Optional[int] = None,
441    cols: Optional[int] = None,
442    no_exit: bool = True,
443) -> HTProcess:
444    """
445    Run a command using the 'ht' tool and return a HTProcess object
446    that can be used to interact with it.
447
448    Args:
449        command: The command to run
450        rows: Number of rows for the terminal size (height)
451        cols: Number of columns for the terminal size (width)
452        no_exit: If True, use the --no-exit flag to keep ht running after subprocess exits
453
454    Returns:
455        An HTProcess instance
456    """
457
458    # Split the command into arguments if it's a string
459    if isinstance(command, str):
460        cmd_args = command.split()
461    else:
462        cmd_args = command
463
464    # Create the ht command with event subscription
465    ht_cmd = [
466        get_ht_binary(),
467        "--subscribe",
468        "init,snapshot,output,resize,pid,exitCode",
469    ]
470
471    # Add size options if specified
472    if rows is not None and cols is not None:
473        ht_cmd.extend(["--size", f"{cols}x{rows}"])
474
475    # Add no-exit option if specified
476    if no_exit:
477        ht_cmd.append("--no-exit")
478
479    # Add separator and the command to run
480    ht_cmd.append("--")
481    ht_cmd.extend(cmd_args)
482
483    # Create a queue for events
484    event_queue = queue.Queue()
485
486    # Create a reader thread to capture ht output
487    def reader_thread(ht_proc, queue_obj, ht_process):
488        while True:
489            line = ht_proc.stdout.readline()
490            if not line:
491                break
492            line = line.strip()
493            if not line:
494                continue
495
496            try:
497                event = json.loads(line)
498                queue_obj.put(event)
499
500                # Store output events separately in the process
501                if event["type"] == "output":
502                    ht_process.output_events.append(event)
503                # Track subprocess exit events
504                elif event["type"] == "exitCode":
505                    ht_process.subprocess_exited = True
506                    if hasattr(ht_process, "subprocess"):
507                        ht_process.subprocess.exit_code = event.get("data", {}).get(
508                            "exitCode", 0
509                        )
510            except json.JSONDecodeError:
511                # Check for non-JSON messages that indicate process state
512                if isinstance(line, str):
513                    if "Process exited" in line:
514                        ht_process.subprocess_exited = True
515                        if hasattr(ht_process, "subprocess") and hasattr(
516                            ht_process.subprocess, "_finished"
517                        ):
518                            ht_process.subprocess._finished = True
519                            ht_process.subprocess.exit_code = 0  # Assume success
520                            ht_process._subprocess_finished = True
521
522                queue_obj.put({"type": "raw", "data": {"text": line}})
523
524    # Log the exact command for debugging
525    import logging
526
527    logger = logging.getLogger(__name__)
528    logger.debug(f"Executing ht command: {' '.join(ht_cmd)}")
529
530    # Launch ht
531    ht_proc = subprocess.Popen(
532        ht_cmd,
533        stdin=subprocess.PIPE,
534        stdout=subprocess.PIPE,
535        stderr=subprocess.PIPE,
536        text=True,
537        bufsize=1,
538    )
539
540    # Create an HTProcess instance
541    process = HTProcess(
542        ht_proc,
543        event_queue,
544        command=" ".join(cmd_args),
545        rows=rows,
546        cols=cols,
547        no_exit=no_exit,
548    )
549
550    # Start the reader thread
551    thread = threading.Thread(
552        target=reader_thread, args=(ht_proc, event_queue, process), daemon=True
553    )
554    thread.start()
555    # Wait briefly for the process to initialize
556    start_time = time.time()
557    while time.time() - start_time < 2:
558        try:
559            event = event_queue.get(block=True, timeout=0.5)
560            if event["type"] == "pid":
561                # Update the subprocess controller
562                pid = event["data"]["pid"]
563                process.subprocess.pid = pid
564                break
565        except queue.Empty:
566            continue
567
568    sleep(0.1)
569    return process

572@contextmanager
573def ht_process(
574    command: str,
575    rows: Optional[int] = None,
576    cols: Optional[int] = None,
577    no_exit: bool = True,
578):
579    """
580    Context manager for HTProcess that ensures proper cleanup.
581
582    Usage:
583        with ht_process("python script.py", rows=10, cols=20) as proc:
584            proc.send_keys(Press.ENTER)
585            snapshot = proc.snapshot()
586            # Process is automatically cleaned up when exiting the context
587
588    Args:
589        command: The command to run
590        rows: Number of rows for the terminal size
591        cols: Number of columns for the terminal size
592        no_exit: Whether to use --no-exit flag (default: True)
593
594    Yields:
595        HTProcess instance with automatic cleanup
596    """
597    proc = run(command, rows=rows, cols=cols, no_exit=no_exit)
598    try:
599        yield proc
600    finally:
601        # Ensure cleanup happens even if an exception occurs
602        try:
603            # Try to terminate subprocess gracefully first
604            if proc.subprocess.pid:
605                proc.subprocess.terminate()
606                proc.subprocess.wait(timeout=2.0)
607        except Exception:
608            # If graceful termination fails, force kill
609            try:
610                if proc.subprocess.pid:
611                    proc.subprocess.kill()
612            except Exception:
613                pass
614
615        try:
616            # Clean up the ht process
617            proc.terminate()
618            proc.wait(timeout=2.0)
619        except Exception:
620            # If ht process won't terminate, force kill it
621            try:
622                proc.kill()
623            except Exception:
624                pass

def clean_ansi_for_html(ansi_text: str) -> str: 
105def clean_ansi_for_html(ansi_text: str) -> str:
106    """
107    Clean ANSI sequences to keep only color/style codes that ansi2html can handle.
108    Removes cursor positioning, screen control, and other non-display sequences.
109    """
110    import re
111
112    # First, normalize \x9b sequences to \x1b[ sequences for consistency
113    ansi_text = ansi_text.replace("\x9b", "\x1b[")
114
115    # Remove cursor positioning sequences like \x1b[1;1H, \x1b[2;3H etc.
116    ansi_text = re.sub(r"\x1b\[\d*;\d*H", "", ansi_text)
117
118    # Remove single cursor positioning like \x1b[H
119    ansi_text = re.sub(r"\x1b\[H", "", ansi_text)
120
121    # Remove screen buffer switching \x1b[?1047h, \x1b[?1047l
122    ansi_text = re.sub(r"\x1b\[\?\d+[hl]", "", ansi_text)
123
124    # Remove scroll region setting \x1b[1;4r
125    ansi_text = re.sub(r"\x1b\[\d*;\d*r", "", ansi_text)
126
127    # Remove save/restore cursor sequences \x1b7, \x1b8
128    ansi_text = re.sub(r"\x1b[78]", "", ansi_text)
129
130    # Remove other terminal control sequences but keep color codes
131    # This removes sequences that don't end with 'm' (which are color codes)
132    ansi_text = re.sub(r"\x1b\[(?![0-9;]*m)[^m]*[a-zA-Z]", "", ansi_text)
133
134    # Remove control characters but preserve \x1b which is needed for ANSI codes
135    # and preserve \r\n for line breaks
136    ansi_text = re.sub(r"[\x00-\x08\x0B-\x1A\x1C-\x1F\x7F-\x9F]", "", ansi_text)
137
138    return ansi_text
