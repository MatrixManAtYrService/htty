325KeyInput = Union[Press, str]

328def key_to_string(key: KeyInput) -> str:
329    if isinstance(key, str):
330        return key
331    elif hasattr(key, "value"):
332        return key.value
333    else:
334        return str(key)
335
336
337def keys_to_strings(keys: Union[KeyInput, List[KeyInput]]) -> List[str]:
338    if not isinstance(keys, list):
339        keys = [keys]
340
341    return [key_to_string(key) for key in keys]
