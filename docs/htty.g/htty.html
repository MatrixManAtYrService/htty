htty - Headless Terminal Python Library

A Python library for terminal automation, providing both synchronous CLI tools and asynchronous library access for subprocess interaction.

This is a fork of ht with enhanced Python integration and reliable subprocess management.

235    def snapshot(self, timeout: float = DEFAULT_SNAPSHOT_TIMEOUT) -> SnapshotResult:
236        """
237        Take a snapshot of the terminal output.
238        """
239        if self.ht_proc.poll() is not None:
240            raise RuntimeError(f"ht process has exited with code {self.ht_proc.returncode}")
241
242        message = json.dumps({"type": "takeSnapshot"})
243        self.logger.debug(f"Taking snapshot: {message}")
244
245        try:
246            if self.ht_proc.stdin is not None:
247                self.ht_proc.stdin.write(message + "\n")
248                self.ht_proc.stdin.flush()
249                self.logger.debug("Snapshot request sent successfully")
250            else:
251                raise RuntimeError("ht process stdin is not available")
252        except BrokenPipeError as e:
253            self.logger.error(f"Failed to send snapshot request: {e}")
254            self.logger.error(f"ht process poll result: {self.ht_proc.poll()}")
255            raise RuntimeError(
256                f"Cannot communicate with ht process (broken pipe). "
257                f"Process may have exited. Poll result: {self.ht_proc.poll()}"
258            ) from e
259
260        time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
261
262        # Process events until we find the snapshot
263        retry_count = 0
264        while retry_count < MAX_SNAPSHOT_RETRIES:
265            try:
266                event = self.event_queue.get(block=True, timeout=SNAPSHOT_RETRY_TIMEOUT)
267            except queue.Empty:
268                retry_count += 1
269                continue
270
271            if event["type"] == "snapshot":
272                data = event["data"]
273                snapshot_text = data["text"]
274                raw_seq = data["seq"]
275
276                # Convert to HTML with ANSI color support
277                html = simple_ansi_to_html(raw_seq)
278
279                return SnapshotResult(
280                    text=snapshot_text,
281                    html=html,
282                    raw_seq=raw_seq,
283                )
284            elif event["type"] == "output":
285                self.output_events.append(event)
286            elif event["type"] == "pid":
287                if self.subprocess_controller.pid is None:
288                    self.subprocess_controller.pid = event["data"]["pid"]
289            elif event["type"] == "exitCode":
290                self.subprocess_exited = True
291                self.subprocess_controller.exit_code = event.get("data", {}).get("exitCode")
292            elif event["type"] == "resize":
293                if "data" in event:
294                    data = event["data"]
295                    if "rows" in data:
296                        self.rows = data["rows"]
297                    if "cols" in data:
298                        self.cols = data["cols"]
299            elif event["type"] == "init":
300                pass
301            else:
302                self.unknown_events.append(event)
303
304        raise RuntimeError(
305            f"Failed to receive snapshot event after {MAX_SNAPSHOT_RETRIES} attempts. "
306            f"ht process may have exited or stopped responding."
307        )

309    def exit(self, timeout: float = DEFAULT_EXIT_TIMEOUT) -> int:
310        """
311        Exit the ht process, ensuring clean shutdown.
312
313        Uses different strategies based on subprocess state:
314        - If subprocess already exited (exitCode event received): graceful shutdown via exit command
315        - If subprocess still running: forced termination with SIGTERM then SIGKILL
316        """
317        self.logger.debug(f"Exiting HTProcess: ht_proc.pid={self.ht_proc.pid}")
318
319        # Check if we've already received the exitCode event
320        if self.subprocess_exited:
321            self.logger.debug("Subprocess already exited (exitCode event received), attempting graceful shutdown")
322            return self._graceful_exit(timeout)
323        else:
324            self.logger.debug("Subprocess has not exited yet, checking current state")
325
326            # Give a brief moment for any pending exitCode event to arrive
327            brief_wait_start = time.time()
328            while time.time() - brief_wait_start < 0.5:  # Wait up to 500ms
329                if self.subprocess_exited:
330                    self.logger.debug("Subprocess exited during brief wait, attempting graceful shutdown")
331                    return self._graceful_exit(timeout)
332                time.sleep(0.01)
333
334            self.logger.debug("Subprocess still running after brief wait, using forced termination")
335            return self._forced_exit(timeout)
336
337    def _graceful_exit(self, timeout: float) -> int:
338        """
339        Graceful exit: subprocess has completed, so we can send exit command to ht process.
340        """
341        # Send exit command to ht process
342        message = json.dumps({"type": "exit"})
343        self.logger.debug(f"Sending exit command to ht process {self.ht_proc.pid}: {message}")
344
345        try:
346            if self.ht_proc.stdin is not None:
347                self.ht_proc.stdin.write(message + "\n")
348                self.ht_proc.stdin.flush()
349                self.logger.debug(f"Exit command sent successfully to ht process {self.ht_proc.pid}")
350                self.ht_proc.stdin.close()  # Close stdin after sending exit command
351                self.logger.debug(f"Closed stdin for ht process {self.ht_proc.pid}")
352            else:
353                self.logger.debug(f"ht process {self.ht_proc.pid} stdin is None, cannot send exit command")
354        except (BrokenPipeError, OSError) as e:
355            self.logger.debug(
356                f"Failed to send exit command to ht process {self.ht_proc.pid}: {e} (process may have already exited)"
357            )
358            pass
359
360        # Wait for the ht process to finish gracefully
361        start_time = time.time()
362        while self.ht_proc.poll() is None:
363            if time.time() - start_time > timeout:
364                # Graceful exit timed out, fall back to forced termination
365                self.logger.warning(
366                    f"ht process {self.ht_proc.pid} did not exit gracefully within timeout, "
367                    f"falling back to forced termination"
368                )
369                return self._forced_exit(timeout)
370            time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
371
372        self.exit_code = self.ht_proc.returncode
373        if self.exit_code is None:
374            raise RuntimeError("Failed to determine ht process exit code")
375
376        self.logger.debug(f"HTProcess exited gracefully: exit_code={self.exit_code}")
377        return self.exit_code
378
379    def _forced_exit(self, timeout: float) -> int:
380        """
381        Forced exit: subprocess may still be running, so we need to terminate everything forcefully.
382        """
383        # Step 1: Ensure subprocess is terminated first if needed
384        if self.subprocess_controller.pid and not self.subprocess_exited:
385            self.logger.debug(f"Terminating subprocess: pid={self.subprocess_controller.pid}")
386            try:
387                os.kill(self.subprocess_controller.pid, 0)
388                self.subprocess_controller.terminate()
389                try:
390                    self.subprocess_controller.wait(timeout=DEFAULT_SUBPROCESS_WAIT_TIMEOUT)
391                    self.logger.debug(f"Subprocess {self.subprocess_controller.pid} terminated successfully")
392                except Exception:
393                    self.logger.warning(
394                        f"Subprocess {self.subprocess_controller.pid} did not terminate gracefully, killing"
395                    )
396                    with suppress(Exception):
397                        self.subprocess_controller.kill()
398            except OSError:
399                self.logger.debug(f"Subprocess {self.subprocess_controller.pid} already exited")
400                pass  # Process already exited
401
402        # Step 2: Force terminate the ht process with SIGTERM, then SIGKILL if needed
403        self.logger.debug(f"Force terminating ht process {self.ht_proc.pid}")
404
405        # Try SIGTERM first
406        try:
407            self.ht_proc.terminate()
408            self.logger.debug(f"Sent SIGTERM to ht process {self.ht_proc.pid}")
409        except Exception as e:
410            self.logger.debug(f"Failed to send SIGTERM to ht process {self.ht_proc.pid}: {e}")
411
412        # Wait for termination
413        start_time = time.time()
414        while self.ht_proc.poll() is None:
415            if time.time() - start_time > timeout:
416                # SIGTERM timeout, try SIGKILL
417                self.logger.warning(
418                    f"ht process {self.ht_proc.pid} did not terminate with SIGTERM within timeout, sending SIGKILL"
419                )
420                try:
421                    self.ht_proc.kill()
422                    self.logger.debug(f"Sent SIGKILL to ht process {self.ht_proc.pid}")
423                except Exception as e:
424                    self.logger.debug(f"Failed to send SIGKILL to ht process {self.ht_proc.pid}: {e}")
425
426                # Wait for SIGKILL to take effect
427                kill_start_time = time.time()
428                while self.ht_proc.poll() is None:
429                    if time.time() - kill_start_time > timeout:
430                        self.logger.error(f"ht process {self.ht_proc.pid} did not respond to SIGKILL within timeout")
431                        break
432                    time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
433                break
434            time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
435
436        self.exit_code = self.ht_proc.returncode
437        if self.exit_code is None:
438            raise RuntimeError("Failed to determine ht process exit code")
439
440        self.logger.debug(f"HTProcess exited via forced termination: exit_code={self.exit_code}")
441        return self.exit_code

533        ht_proc: subprocess.Popen[str],
534        queue_obj: queue.Queue[dict[str, Any]],
535        ht_process: HTProcess,
536        thread_logger: logging.Logger,
537    ) -> None:
538        thread_logger.debug(f"Reader thread started for ht process {ht_proc.pid}")
539
540        while True:
541            if ht_proc.stdout is None:
542                thread_logger.warning(f"ht process {ht_proc.pid} stdout is None, exiting reader thread")
543                break
544
545            line = ht_proc.stdout.readline()
546            if not line:
547                thread_logger.debug(f"ht process {ht_proc.pid} stdout closed, exiting reader thread")
548                break
549
550            line = line.strip()
551            if not line:
552                continue
553
554            try:
555                event = json.loads(line)
556                thread_logger.debug(f"ht event: {event}")
557                queue_obj.put(event)
558
559                if event["type"] == "output":
560                    ht_process.output_events.append(event)
561                elif event["type"] == "exitCode":
562                    thread_logger.debug(

 73
 74    def __init__(self, pid: Optional[int] = None):
 75        self.pid = pid
 76        self.exit_code: Optional[int] = None
 77        self._termination_initiated = False
 78
 79    def poll(self) -> Optional[int]:
 80        """Check if the subprocess is still running."""
 81        if self.pid is None:
 82            return self.exit_code
 83        try:
 84            os.kill(self.pid, 0)
 85            return None  # Process is still running
 86        except OSError:
 87            return self.exit_code  # Process has exited
 88
 89    def terminate(self) -> None:
 90        """Terminate the subprocess."""
 91        if self.pid is None:
 92            raise RuntimeError("No subprocess PID available")
 93        try:
 94            self._termination_initiated = True
 95            os.kill(self.pid, signal.SIGTERM)
 96        except OSError:
 97            pass  # Process may have already exited
 98
 99    def kill(self) -> None:
100        """Force kill the subprocess."""
101        if self.pid is None:
102            raise RuntimeError("No subprocess PID available")
103        try:
104            self._termination_initiated = True
105            os.kill(self.pid, signal.SIGKILL)
106        except OSError:
107            pass  # Process may have already exited
108
109    def wait(self, timeout: Optional[float] = 5.0) -> Optional[int]:
110        """
111        Wait for the subprocess to finish.
112
113        Args:
114            timeout: Maximum time to wait (in seconds). Defaults to 5.0 seconds.
115
116        Returns:
117            The exit code of the subprocess, or None if timeout reached
118        """
119        if self.pid is None:
120            raise RuntimeError("No subprocess PID available")
121
122        start_time = time.time()
123        while True:
124            try:
125                os.kill(self.pid, 0)  # Check if process is still running
126
127                # Check timeout
128                if timeout is not None and (time.time() - start_time) > timeout:
129                    return None  # Timeout reached
130
131                time.sleep(DEFAULT_SLEEP_AFTER_KEYS)
132            except OSError:
133                # Process has exited
134                # Try to get the actual exit code
135                try:
136                    pid_result, status = os.waitpid(self.pid, os.WNOHANG)
137                    if pid_result == self.pid:
138                        if os.WIFEXITED(status):
139                            self.exit_code = os.WEXITSTATUS(status)
140                        elif os.WIFSIGNALED(status):
141                            signal_num = os.WTERMSIG(status)
142                            self.exit_code = 128 + signal_num
143                        else:
144                            self.exit_code = 1
145                    else:
146                        # Process was already reaped, use stored exit code or default
147                        if self.exit_code is None:
148                            self.exit_code = 0 if not self._termination_initiated else 137
149                except OSError:
150                    # Couldn't get exit code, use reasonable default
151                    if self.exit_code is None:
152                        self.exit_code = 0 if not self._termination_initiated else 137
153
154                return self.exit_code
